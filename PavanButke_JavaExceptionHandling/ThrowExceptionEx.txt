package core_javaBeginner;

public class ThrowException {
	

		public static void validVoter(int age){
			if(age<18)
	            try{
	                throw new Exception();
	            }catch(Exception e){
	                System.out.println("Too young to Vote! ");
	            }
			else
			   System.out.println("Do your Duty , Vote  Everytime! ");
		}
		public static void main (String[] args) {
			validVoter(20);
			validVoter(10);
		}
	}

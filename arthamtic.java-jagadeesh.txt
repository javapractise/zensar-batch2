package addtwonumbers;

public class sumoftwo {

	public static void main(String[] args) {
		int n1=20,n2=35,add,sub,div,mul,rem;
		
		add=n1+n2;
		sub=n1-n2;
		div=n1/n2;
		mul=n1*n2;
		rem=n1|n2;
		
		
		System.out.println("add of these numbers:"+add);
		System.out.println("sub of these numbers:"+sub);
		System.out.println("div of these numbers:"+div);
		System.out.println("mul of these numbers:"+mul);
		System.out.println("rem of these numbers:"+rem);

	}

}

package partice;

import static java.lang.System.*;

public class Threadpriority extends Thread{
    public void run(){
        System.out.println("inside the run() method");
    }

    public static void main(String[] args) {
        Threadpriority t1=new Threadpriority();
        Threadpriority t2=new Threadpriority();
        Threadpriority t3=new Threadpriority();
        System.out.println("priority of the thread t1 is:"+ t1.getPriority());
        System.out.println("priority of the thread t2 is:"+ t2.getPriority());
        System.out.println("priority of the thread t3 is:"+ t3.getPriority());
        t1.setPriority(1);
        t2.setPriority(2);
        t3.setPriority(3);
        out.println("Priority of the thread t1 is:"+t1.getPriority());

        out.println("Priority of the main thread is:"+Thread.currentThread().getPriority() );

        out.println("Priority of the main thread is:"+Thread.currentThread().getPriority() );

    }
}

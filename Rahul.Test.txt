package Test;
import java.util.*;
public class Exam {
public static void main(String[] args) {
	Scanner s = new Scanner(System.in);
	String old = s.next().toLowerCase();
	String neW = s.next().toLowerCase();
	String res = "";
	
	if(old.equalsIgnoreCase(neW)) {
		System.out.println("0");
	}
	else {
		if(old.length() == neW.length()) {
			for(int i=0;i<old.length();i++) {
				if(neW.contains(""+old.charAt(i))) {
					res += old.charAt(i);
				}
			}
		}
		if(res.equalsIgnoreCase(old)){
			System.out.println("1");
		}
		else {
		System.out.println("0");
	}
	}
	s.close();
}
}

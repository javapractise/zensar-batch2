

import java.util.*;
import java.io.*;
class Test {
    public int func(int[] arr, int l)
{
    int r=0;
    int s=1;
    for(int i=0;i<l; i++)
    {
        s=s*arr[i];
    }
    if(s>0)
    {
        r=1;
    }
    else if(s<0)
    {
        r=-1;
    }
    return r;
    
}
    public static void main(String[] args) {
        Scanner sc= new Scanner(System.in);
        int s=0;
        System.out.println("Enter size of array: ");
        s=sc.nextInt();
        int arr[]= new int[s];
        for(int i=0; i<s; i++)
        {
            System.out.println("Enter value at "+i+"th position: ");
            arr[i]= sc.nextInt();
        }
        Test ob=new Test();
        int ans= ob.func(arr, s);
        System.out.println("Ans= "+ans); 
    }
}

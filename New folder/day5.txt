SINGLE INHERITANCE:-

public class Vinay {
    void study()
    {
        System.out.println("studying");
    }
}
class Vicky extends Vinay{
    void play()
    {
        System.out.println("playing");
    }
        }
        class Single{
    public static void main(String[] args) {
        Vicky p=new Vicky();
        p.study();
        p.play();

    }
}

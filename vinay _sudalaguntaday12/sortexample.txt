import java.util.*;
class Sort1{
public static void main(String args[]){

	ArrayList<String> al=new ArrayList<String>();
	al.add("VINAY");
	al.add("VICKY");
	al.add("SUMANTH");
	al.add("AKASH");



	Collections.sort(al); 
	Iterator itr=al.iterator();
	while(itr.hasNext()){
	System.out.println(itr.next());
	}
}
}

import java.util.*;
public class WrapperClass{
	public static  void main(String args[]){
		ArrayList a1=new ArrayList();
		a1.add(Integer.valueOf(11));
		a1.add(Integer.valueOf(121));
		a1.add(212);
		Collections.sort(a1);
		Iterator itr=a1.iterator();
		while(itr.hasNext()){
			System.out.println(itr.next());
		}
	}
}

output:-
11
121
212

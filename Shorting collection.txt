import java.util.*;  
class TestSort1{  
public static void main(String args[]){  
  
ArrayList<String> al=new ArrayList<String>();  
al.add("NAGARAJ");  
al.add("SIVAYYA");  
al.add("NARESH");  
al.add("KAMESWAR");  
  
Collections.sort(al);  
Iterator itr=al.iterator();  
while(itr.hasNext()){  
System.out.println(itr.next());  
 }  
}  
}  

OUTPUT::NAGARAJ
SIVAYYA
NARESH
KAMESWAR

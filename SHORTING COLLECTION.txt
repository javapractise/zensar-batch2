import java.util.*;  
class TestSort1{  
public static void main(String args[]){  
  
ArrayList<String> al=new ArrayList<String>();  
al.add("RAGHAV");  
al.add("SIVA");  
al.add("HARESH");  
al.add("SANTHOSH");  
  
Collections.sort(al);  
Iterator itr=al.iterator();  
while(itr.hasNext()){  
System.out.println(itr.next());  
 }  
}  
}  

OUTPUT::HARESH
RAGHAV
SANTHOSH
SIVA

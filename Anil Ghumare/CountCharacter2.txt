public class Main {    
    public static void main(String[] args) {    
        String text = "How many characters are there in text string";    
        int count = 0;    
            
        //Counts each character except space    
        for(int i = 0; i < text.length(); i++) {    
            if(text.charAt(i) != ' ')    
                count++;    
        }    
            
        //Displays the total number of characters present in the given string    
        System.out.println("Total number of characters in a text string: " + count);    
    }    
}

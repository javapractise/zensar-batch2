import java.util.Iterator;
import java.util.Vector;
public class VectorExa{
	public static void main(String args[]){
		Vector<String>v= new Vector<String>();
		v.add("Jayshri");
		v.add("Vani");
		v.add("Shivani");
		v.add("Manasi");
		Iterator<String>itr=v.iterator();
		while(itr.hasNext()){
			System.out.println(itr.next());
		}
	}
}

/*
Output:
Jayshri
Vani
Shivani
Manasi
*/

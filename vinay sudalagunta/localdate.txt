
import java.time.*;  
public class LocalDateExample {  
  public static void main(String[] args) {  
    LocalDate date = LocalDate.of(2010, 11, 16);  
    LocalDateTime datetime = date.atTime(1,43,9);      
    System.out.println(datetime);   
  }  
}  

output:-
2010-11-16T01:43:09

interface Addable{  
    int add(int a,int b);  
}  
  
public class LambdaExpression {  
    public static void main(String[] args) {  
          
       
        Addable ad1=(a,b)->(a+b);  
        System.out.println(ad1.add(105,293));  
        Addable ad2=(int a,int b)->{  
                            return (a+b);   
                            };  
        System.out.println(ad2.add(180,290));  
    }  
}  

output:-
398
470

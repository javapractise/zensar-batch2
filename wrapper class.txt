1.converting primitive into wrapper object 

public class Main
 {  
   public static void main(String args[])
   {  
	int num=100;  
	Integer S =Integer.valueOf(num);  

	System.out.println(num+ " "+ S);  
   }
}

Output:

100 100


2.Converting wrapper class object into primitive 


public class Main
 {  
   public static void main(String args[])
   {  
	Integer I = new Integer(77);  
	int num = I.intValue();

	System.out.println(num+ " "+ I);  
   }
}


Output:
  77 77


3.custom wrapper class 


class Wrapper3
{  
private int i;  
Wrapper3(int i)
{  
this.i=i;  
}  
public int getValue(){  
return i;  
}  
public void setValue(int i){  
this.i=i;  
}  
public String toString() {  
  return Integer.toString(i);  
}  
}  
public class Main 
{  
public static void main(String[] args)
{  
Wrapper3 j=new Wrapper3(21);  
System.out.println(j);  
}
}  


output:
 21

public class Main{

byte b=11;

Float floatobj=f;  
Double doubleobj=d;  
Character charobj=c;

//Printing objects

//Unboxing: Converting Objects to Primitives  
byte bytevalue=byteobj;  
short shortvalue=shortobj;  
int intvalue=intobj;  
long longvalue=longobj;  
float floatvalue=floatobj;  
double doublevalue=doubleobj;

//Printing primitives
